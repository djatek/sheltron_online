  <script>

    // ============ Simple tab/controller logic ============

    const tabs = document.querySelectorAll('.tab');

    const panels = document.querySelectorAll('.panel');

    const topButtons = document.querySelectorAll('.menu [data-target]');


    function show(id){

      panels.forEach(p=>{ p.hidden = (p.id !== id.replace('#','')); });

      tabs.forEach(t=>{ t.classList.toggle('active', t.getAttribute('aria-controls') === id.replace('#','')); t.setAttribute('aria-selected', t.classList.contains('active')); });

      document.getElementById('app-window').scrollIntoView({behavior:'smooth', block:'start'});

    }


    tabs.forEach(t=> t.addEventListener('click', ()=> show('#'+t.getAttribute('aria-controls'))));

    tabs.forEach(t=> t.addEventListener('keydown', (e)=>{ if(e.key==='Enter' || e.key===' '){ e.preventDefault(); show('#'+t.getAttribute('aria-controls')); }}));


    topButtons.forEach(b=> b.addEventListener('click', ()=> show(b.dataset.target)));


    // Default hash routing (optional):

    if(location.hash && document.querySelector(location.hash)){

      show(location.hash);

    }


    // Year in footer

    document.getElementById('year').textContent = new Date().getFullYear();

  </script>
